Проверяем строку на содержание только цифр:

local str = "*123#"
if type(tonumber(str)) == 'number' then
   print('Only numbers ')
else
   print('Not only numbers ')
end
или так:
local str = "123"
if (string.match(str,"[^%w]") == nil) then
  print('Only numbers ')
else
  print('Not only numbers ')
end

строку в таблицу

 

local text = "bb115bed-ac46-4d43-b2d3-00d923e225af;9f9ea88e-45ce-4fdc-9436-ec686c795215"
local textArray = {}

for str in text:gmatch("([^(,;) ]+)") do
 table.insert(textArray, str)
end

for _, value in pairs(textArray) do
  print("--> "..value)
end
print("==> "..#textArray)

Удаление

Задача 1:  удалить у строки первые 3 символа 

str = "12345678901234567890"
out = ""
for i = 3 + 1 , #str do
   local tmp = str:sub(i, i)
   out = out .. tmp
end
print(out)  -- 45678901234567890

или 

str = "12345678901234567890"
out = ""
for i = 3 + 1, string.len(str) do
   local tmp = string.sub(str, i, i)
   out = out .. tmp
end
print(out)  -- 45678901234567890
